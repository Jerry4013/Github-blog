## git 添加子模块 

git submodule add module 

## git 删除子模块 

git submodule deinit module 
git rm module 
rm -rf .git/modules/module 

## git 子模块初始化及更新 

git submodule init 
git submodule update 

## git 查看远程仓库地址 

git remote -v 

## git 更换远程仓库 

git remote remove origin 
git remote add origin your@addr
